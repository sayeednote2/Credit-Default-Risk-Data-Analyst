    [Total Customers],
    fact_applications[DATASET] = "train"
)

Test Customers =
CALCULATE(
    [Total Customers],
    fact_applications[DATASET] = "test"
)

Default Count =
CALCULATE(
    DISTINCTCOUNT(fact_applications[SK_ID_CURR]),
    fact_applications[TARGET] = 1
)

    DISTINCTCOUNT(fact_applications[SK_ID_CURR]),
    fact_applications[BASE_RISK_BAND] IN {"High", "Very High"}

Total Applications = COUNTROWS(fact_applications)

Bureau Loan Count = SUM(fact_bureau[BUREAU_LOAN_COUNT])

Bureau Debt Sum = SUM(fact_bureau[BUREAU_DEBT_SUM])

Bureau Overdue Sum = SUM(fact_bureau[BUREAU_OVERDUE_SUM])

Previous Applications = SUM(fact_prev_applications[PREV_APP_COUNT])

Refused Prev Apps = SUM(fact_prev_applications[PREV_REFUSED_COUNT])

Refusal Rate = DIVIDE([Refused Prev Apps], [Approved Prev Apps] + [Refused Prev Apps], 0)

-- Model monitoring
Scored Customers = DISTINCTCOUNT(fact_model_predictions[SK_ID_CURR])

Train Scored Customers =
CALCULATE(
    [Scored Customers],
    fact_model_predictions[DATASET] = "train"
)

Test Scored Customers =
CALCULATE(
    [Scored Customers],
    fact_model_predictions[DATASET] = "test"
)

Predicted Defaults =
CALCULATE(
    DISTINCTCOUNT(fact_model_predictions[SK_ID_CURR]),
    fact_model_predictions[PRED_LABEL] = 1
)

Train Predicted Defaults =
CALCULATE(
    [Predicted Defaults],
    fact_model_predictions[DATASET] = "train"
)

Predicted Default Rate = DIVIDE([Predicted Defaults], [Scored Customers], 0)

Train Predicted Default Rate = DIVIDE([Train Predicted Defaults], [Train Scored Customers], 0)

Avg Pred Score = AVERAGE(fact_model_predictions[PRED_SCORE])

High Risk Customers =
CALCULATE(
    DISTINCTCOUNT(fact_model_predictions[SK_ID_CURR]),
    fact_model_predictions[PRED_RISK_BAND] IN {"High", "Very High"}
)

High Risk Share = DIVIDE([High Risk Customers], [Scored Customers], 0)

Actual Defaults (Train) =
CALCULATE(
    DISTINCTCOUNT(fact_model_predictions[SK_ID_CURR]),
    fact_model_predictions[DATASET] = "train",
    fact_model_predictions[TARGET] = 1
)

Observed Train Default Rate = DIVIDE([Actual Defaults (Train)], [Train Scored Customers], 0)

True Positives =
CALCULATE(
    DISTINCTCOUNT(fact_model_predictions[SK_ID_CURR]),
    fact_model_predictions[DATASET] = "train",
    fact_model_predictions[TARGET] = 1,
    fact_model_predictions[PRED_LABEL] = 1
)

False Positives =
CALCULATE(
    DISTINCTCOUNT(fact_model_predictions[SK_ID_CURR]),
    fact_model_predictions[DATASET] = "train",
    fact_model_predictions[TARGET] = 0,
    fact_model_predictions[PRED_LABEL] = 1
)

True Negatives =
CALCULATE(
    DISTINCTCOUNT(fact_model_predictions[SK_ID_CURR]),
    fact_model_predictions[DATASET] = "train",
    fact_model_predictions[TARGET] = 0,
    fact_model_predictions[PRED_LABEL] = 0
)

False Negatives =
CALCULATE(
    DISTINCTCOUNT(fact_model_predictions[SK_ID_CURR]),
    fact_model_predictions[DATASET] = "train",
    fact_model_predictions[TARGET] = 1,
    fact_model_predictions[PRED_LABEL] = 0
)

Model Precision = DIVIDE([True Positives], [True Positives] + [False Positives], 0)

Model Recall = DIVIDE([True Positives], [True Positives] + [False Negatives], 0)

Model Accuracy =
DIVIDE(
    [True Positives] + [True Negatives],
    [True Positives] + [True Negatives] + [False Positives] + [False Negatives],
    0
)

Model F1 =
VAR PrecisionValue = [Model Precision]
VAR RecallValue = [Model Recall]
RETURN DIVIDE(2 * PrecisionValue * RecallValue, PrecisionValue + RecallValue, 0)
